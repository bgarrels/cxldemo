327       }, this);

334          dynamic:true,

342       this.createConfig();
343       //Context Menu
344       this.initContextMenu();
345       // Check if whe have to load a external file
346       if (this.autoLoad) {
347        if (typeof this.autoLoad !== 'object')  this.autoLoad = {url: this.autoLoad};
348        if (this.autoLoad['nocache']==undefined) this.autoLoad['nocache'] = this.nocache;

393   /**
394    * Start resize of an element, it will become active element
395    * @param {Element} element The element to resize
396    * @param {Boolean} select Should element be selected (default true) 
397    */
398   visualResize : function(element,select) {
399     var cmp= select===false ? element : this.selectElement(element);
400     if (!cmp) return;
401     var own = this.getContainer(cmp.ownerCt);
402     var layout = own.codeConfig ? own.codeConfig.layout : null;
403     if (layout=='fit') {
404      this.fireEvent('error','visualResize','Cannot resize element within fit layout');
405     } else {
406       //Incase of absolute layout enable movement within element
407       if (layout=='absolute') {
408         this.resizeLayer.resizer.dd.unlock();
409         this.resizeLayer.resizer.dd.constrainTo(own.body);
410       } else {
411         this.resizeLayer.resizer.dd.lock();
412       }
413       this.resizeLayer.setBox(cmp.el.getBox());
414       this.resizeLayer.show();
415     }
416   },

418   /**
419    * Move a element within absolute layout based on drag event of the resize layer
420    */
421   moveElement : function() {
422     var cmp = this.activeElement;
423     var pos = this.resizeLayer.getXY();
424     var oPos = this.getContainer(cmp.ownerCt).body.getXY();
425     this.markUndo();
426     cmp.codeConfig.x = pos[0] - oPos[0];
427     cmp.codeConfig.y = pos[1] - oPos[1];
428     this.redrawElement();

431   /**
432    * resize an element based on the resize of resizelayer
433    * @param {} r
434    * @param {int} w New width of element
435    * @param {int] h New height of element
436    */
437   resizeElement : function(r,w,h) {
438     var cmp = this.activeElement;
439     var s = cmp.el.getSize();
440     this.markUndo();
441     if (s.width != w) {
442       cmp.codeConfig.width = w;
443       delete cmp.codeConfig.columnWidth;
444     }
445     if (s.height !=h) {
446       cmp.codeConfig.height = h;
447       delete cmp.codeConfig.autoHeight;
448     }
449     this.redrawElement();
450   },

452   /**
453    * Remove an element
454    * @param {Element} source The element to remove
455    * @param {Boolean\String} internal When true no remove event and redraw is fired,
456    * when "noundo" no undo is created (defaults false)
457    * @return {Boolean} Indicator telling element was removed
458    */
459   removeElement : function(source,internal) {
460     if (!source) return false;
461     var own = this.getContainer(source.ownerCt);
462     if (!internal) this.markUndo();
463     for (var i=0;i<own.items.length;i++) {
464       if (own.items.items[i]==source) {
465         own.codeConfig.items.splice(i,1);
466         own.remove(source,true);
467         if (own.codeConfig.items.length==0) delete own.codeConfig.items;
468         if (!internal || internal=="noundo") {
469           this.redrawElement(own,this.getJsonId(this.activeElement==source ? own : this.activeElement));
470           this.fireEvent('remove');
471         } else {
472 
473           this.redrawContainer = true;
474         }
475         return true;
476       }
477     }
478     return false;
479   },

491   /**
492    * Mark a configuration for undo
493    */
494   markUndo : function() {
495     while (this.undoHistory.length>this.undoHistoryMark) this.undoHistory.pop();
496     this.undoHistory.push({config : this.encode(this.getConfig(),0,true),activeId: this.getJsonId()});
497     while (this.undoHistory.length > this.undoHistoryMax) this.undoHistory.shift();
498     this.undoHistoryMark = this.undoHistory.length;
499     this.menuUpdate();
500   },
501 
502   /**
503    * Undo a configuration
504    */
505   undo : function(){
506     if (this.undoHistoryMark>0) {
507       if (this.undoHistoryMark==this.undoHistory.length) {
508         //Make sure whe have point to recover incase of redo
509         this.undoHistory.push({config:this.encode(this.getConfig(),0,true),activeId : this.getJsonId()});
510         this.undoHistoryMark = this.undoHistory.length-1;
511       }
512       this.undoHistoryMark--;
513       var undoObj =this.undoHistory[this.undoHistoryMark];
514       this.setConfig(undoObj.config,true);
515       this.selectElement(undoObj.activeId);
516       this.menuUpdate();
517     }
518   },
519 
520   /**
521    * Redo an undo configuration
522    */
523   redo : function(){
524     if (this.undoHistory.length>this.undoHistoryMark+1) {
525       this.undoHistoryMark++;
526       var redoObj = this.undoHistory[this.undoHistoryMark];
527       this.setConfig(redoObj.config,true);
528       this.selectElement(redoObj.activeId);
529       this.menuUpdate();
530     }

533   /**
534    * Append the config to the element
535    * @param {Element} el The element to which the config would be added
536    * @param {Object} config The config object to be added
537    * @param {Boolean} select Should item be selected
538    * @param {String} dropLocation The operation to perform
539    * @param {Object} source The source used to perform operation
540    * @param {Object} extraConfig A extra config that should be added to config
541    * @return {Component} The component added
542    */
543   appendConfig : function (el,config,select,dropLocation,source,extraConfig){
544     if (!el) return false;
545     this.markUndo();
546 
547     //Custom function for adding stuff to a container
548     var add =  function(src,comp,at,before){
549       if(!src.items) src.initItems();
550       var pos = src.items.length;
551       for (var i=0;i<src.items.length;i++) {
552         if (src.items.items[i]==at) {
553           pos =  (before) ? i : i+1;
554           i=src.items.length;
555         }
556       }
557       if (!src.codeConfig.items || !(src.codeConfig.items instanceof Array))
558           src.codeConfig.items =  [];
559       delete src.codeConfig.html; //items and html go not together in IE
560       if (pos>=src.codeConfig.items.length)
561         src.codeConfig.items.push(comp)
562       else
563        src.codeConfig.items.splice(pos, 0, comp);
564     }.createDelegate(this);
565 
566 
567     if (typeof config == 'function') {
568       config.call(this,function(config) {
569         this.appendConfig(el,config,true,dropLocation,source,extraConfig);
570       }.createDelegate(this),this);
571     } else {
572      //Get the config of the items
573      var ccmp,cmp= this.getDesignElement(el,true);
574      var items = this.editable(Ext.applyIf(this.clone(config),extraConfig || {}));
575      //Transform a config to prevent errors
576      var transformAppend = function(cmp,last){
577        if (!cmp) return;
578        //Transform form to layout if allready contained
579        if (items.xtype=='form' && cmp instanceof Ext.form.FormPanel) {
580          delete items.xtype;
581          items.layout='form';
582        }
583        transformAppend(this.getContainer(cmp.ownerCt),cmp);
584      }.createDelegate(this);
585      transformAppend(cmp);
586      //Find the container that should be changed
587      ccmp = this.getContainer(cmp);
588      switch (dropLocation) {
589        case 'abovecode' :
590        case 'belowcode' :
591          ccmp = this.isContainer(cmp) ? this.getContainer(cmp.ownerCt) : ccmp;
592        case 'appendcode' :
593          this.removeElement(source,true);
594          add(ccmp,items,cmp,dropLocation=='abovecode');
595          break;
596        case  'appendafter' :
597          add(ccmp,items,cmp,false);
598          break;
599        case 'appendbefore' :
600          add(ccmp,items,cmp,true);
601          break;
602        case 'moveafter' :
603          this.removeElement(source,true);
604          add(ccmp,items,cmp,false);
605          break;
606        case 'movebefore' :
607          this.removeElement(source,true);
608          add(ccmp,items,cmp,true);
609          break;
610        case 'move' :
611          this.removeElement(source,true);
612          add(ccmp,items);
613          break;
614        default :
615         add(ccmp,items);
616      }
617      this.modified = true;
618      this.fireEvent('add');
619      this.redrawElement(ccmp,items[this.jsonId]);
620     }
621     return false;
622   },

624   /**
625    * Create the codeConfig object and apply it to the field
626    */
627   createConfig : function() {
628     if (this.container.items && this.container.items.first()) {
629        var items = [];
630        while (this.container.items.first()) {
631           items.push(this.container.items.first());
632           this.container.items.remove(this.container.items.first());
633        }
634        //Re create a panel with items from config editable root
635        var config = { 'border' : false, 'layout' : this.container.getLayout(),'items' : this.editable(items)};
636        config[this.jsonId]=Ext.id();
637        var el = this.container.add(config);
638        el.codeConfig = config;
639     }
640   },

659      Ext.Ajax.request({
660       url: url,
661       method : 'GET',
662       callback: function(options, success, response){
663         if (success) {
664          this.setConfig(response.responseText,false);
665          this.modified = false;
666         } else {
667          if (!this.fireEvent('loadfailed',url,response))
668             Ext.Msg.alert('Failure','Failed to load url :' + url);
669         }
670         if (this.loadMask && this.container.ownerCt)
671            this.container.ownerCt.el.unmask();
672       },
673       scope: this
674      });

686  /**
687    * Function require is used to checks if the required files are loaded
688    * when not the files are loaded. Loaded files are added to required_js and required_css
689    * @param {Mixed} packages An array or ; seperated string with packages to load
690    * @param {Mixed} options An object with options to used, when string then basedir is assumed
691    * options[basedir] The default directory to use for all files
692    * options[cssdir] The directory to be used for stylesheets
693    * options[async] When set to true required will return directly
694    * options[callback] Callback function after all required files are loaded
695    * options[reload] When reload is set packages are reloaded
696    * options[nocache] When set the object caching is turned off
697    * @return {Object} a object with keys js and css contain an array with full path of items
698    */
699   require : function(packages,options){
700     var ret = Ext.ux.guid.plugin.Designer.superclass.require.call(this,packages,options);
701     if (ret.js.length || ret.css.length) {
702       var nocache = (options && options.nocache!=undefined) ? options.nocache : this.nocache;
703       var cfg = (this.container.items && this.container.items.items.length==1) ?
704              this.getConfig(this.container.items.items[0]) : {};
705       //Check if root is a array with more then one element, if so skip
706        if (!cfg.json) cfg.json = {};
707
