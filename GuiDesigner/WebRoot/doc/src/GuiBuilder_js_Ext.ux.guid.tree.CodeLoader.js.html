 20 //Register name spaces used
 21 Ext.namespace('Ext.ux.guid.tree');
 22 
 23 /**
 24  * Constructor of the Codeloader
 25  */
 26 Ext.ux.guid.tree.CodeLoader = function(designer,config) {
 27    Ext.apply(this, config);
 28    this.designer = designer;
 29    Ext.tree.TreeLoader.superclass.constructor.call(this);
 30 };
 31 
 32 /**
 33  * CodeLoader extends Ext.tree.TreeLoader to enable loading of the designer Components
 34  * The name of the element is created based on components data
 35  */
 36 Ext.extend(Ext.ux.guid.tree.CodeLoader, Ext.tree.TreeLoader, {
 37  // @private jsonId is the id used to store the original code
 38  jsonId : '__JSON__',
 39 
 40   /**
 41    * Convert an element into a self explaining name
 42    * @param c {component} The component for which to create a name
 43    */
 44   elementToText : function(c) {
 45       var txt = [];
 46       c = c || {};
 47       if (c[this.jsonId + 'xtype']) {
 48         var v = c[this.jsonId + 'xtype'];
 49         txt.push(typeof(v)=='object' ? v.display || v.value : v);
 50       } else if (c.xtype) { txt.push(c.xtype); }
 51       if (c.fieldLabel) { txt.push('[' + c.fieldLabel + ']'); }
 52       if (c.boxLabel)   { txt.push('[' + c.boxLabel + ']'); }
 53       if (c.layout)     { txt.push('<i>' + c.layout + '</i>'); }
 54       if (c.title)      { txt.push('<b>' + c.title + '</b>'); }
 55       if (c.text)       { txt.push('<b>' + c.text + '</b>'); }
 56       if (c.region)     { txt.push('<i>(' + c.region + ')</i>'); }
 57       return (txt.length == 0 ? "Element" : txt.join(" "));
 58   },
 59 
 60  // @private load is called when content of tree should be reloaded
 61  load : function(node, callback){
 62     node.beginUpdate();
 63     while(node.firstChild) node.removeChild(node.firstChild);
 64      if(this.doLoad(node,this.designer.getConfig())){
 65        if(typeof callback == "function") callback();
 66      }
 67     node.endUpdate();
 68   },
 69 
 70   // @private the interal loop used to go through the data and build a tree
 71   doLoad : function(node,data){
 72     if(data){
 73       if (data instanceof Array) {
 74         for(var i=0;i<data.length;i++) this.doLoad(node,data[i]);
 75       } else if (!this.designer.isEmpty(data)) {
 76    			var isContainer = this.designer.isContainer(this.designer.findByJsonId(data[this.jsonId]));
 77         var cs = {
 78              text: this.elementToText(data),
 79              cls: isContainer ? 'folder' : 'file' ,
 80              leaf : isContainer ? false : true,
 81              jsonId : data[this.jsonId]
 82         };
 83         var cn = node.appendChild(new Ext.tree.TreeNode(cs));
 84         if (data.items) {
 85           for(var i = 0, len = data.items.length; i < len; i++){
 86             this.doLoad(cn,data.items[i]);
 87           }
 88         }
 89       }
 90     }
 91     return !!data;
 92   }
 93 });
