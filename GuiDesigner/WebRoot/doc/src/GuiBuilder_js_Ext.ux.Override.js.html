 21 /**
 22  * Check if ExtJS version is between from and to version
 23  * @param {String} fromVersion The version which is at least required
 24  * @param {String} toVersion The version which is used as end (default ExtJS version)
 25  * @return {Boolean} True when ExtJS version is between from and to version
 26  */
 27 Ext.isVersion = function(fromVersion,toVersion) {
 28   if (fromVersion instanceof Array) {
 29     toVersion = formVersion[1];
 30     fromVersion = formVersion[0];
 31   }
 32   var getVersion = function(ver) {
 33     var major = ver.match(/^(\d)\.\d/);
 34     major = major ? major[1] : 0;
 35     var minor = ver.match(/^\d\.(\d)/);
 36     minor = minor ? minor[1] : 0;
 37     var revision = ver.match(/^\d\.\d\.(\d)/);
 38     revision = revision ? revision[1] : 0;
 39     return (major*1) + (minor*0.1) + (revision*0.001);
 40   }
 41   var f = getVersion(fromVersion || Ext.version);
 42   var t = getVersion(toVersion || Ext.version);
 43   var e = getVersion(Ext.version);
 44   return (e>=f && e<=t);
 45 }

 47 /**
 48  * Override a class when the version matches according to isVersion
 49  * @param {Class} cl The class to override
 50  * @param {Object} obj The object containing the function to be overwritten
 51  * @param {String} fromVersion The version which is at least required
 52  * @param {String} toVersion The version which is used as end (default ExtJS version)
 53  */
 54 Ext.overrideIf = function(cl,obj,fromVersion,toVersion) {
 55   if (Ext.isVersion(fromVersion,toVersion)) {
 56     return Ext.override(cl,obj);
 57   }
 58   return cl;
 59 };

 61 /**
 62  * Apply object to source when version matches
 63  * @param {Object} source The source object
 64  * @param {Object} obj The object to apply
 65  * @param {String} fromVersion The version which is at least required
 66  * @param {String} toVersion The version which is used as end (default ExtJS version)
 67  */
 68 Ext.applyVersion = function(source,obj,fromVersion,toVersion) {
 69   if (Ext.isVersion(fromVersion,toVersion)) {
 70     return Ext.apply(source,obj);
 71   }
 72   return source;
 73 };
 74 
 75 /**
 76  * Apply object to source when version matches
 77  * @param {Object} source The source object
 78  * @param {Object} obj The object to apply
 79  * @param {String} fromVersion The version which is at least required
 80  * @param {String} toVersion The version which is used as end (default ExtJS version)
 81  */
 82 Ext.applyIfVersion = function(source,obj,fromVersion,toVersion) {
 83   if (Ext.isVersion(fromVersion,toVersion)) {
 84     return Ext.applyIf(source,obj);
 85   }
 86   return source;
 87 };

 89 /**
 90  * Override Ext.Panel so that scope of keymap is always set to object when not set,
 91  * instead of window
 92  */
 93 Ext.overrideIf(Ext.Panel,{
 94    // private
 95     getKeyMap : function(){
 96       if(!this.keyMap){
 97         if(Ext.isArray(this.keys)){
 98           for(var i = 0, len = this.keys.length; i < len; i++){
 99             this.keys[i].scope = this.keys[i].scope || this;
100           }
101         } else if (this.keys && !this.keys.scope) this.keys.scope = this;
102         this.keyMap = new Ext.KeyMap(this.el, this.keys);
103       }
104       return this.keyMap;
105     }
106 },'2.0');

109 /**
110  * Override Ext.FormPanel so that in case whe create a form without items from a json
111  * it still has a item list.
112  */
113 Ext.overrideIf(Ext.FormPanel, {
114     // private
115     initFields : function(){
116         //BEGIN FIX It can happend that there is a form created without items (json)
117         this.initItems();
118         //END FIX
119         var f = this.form;
120         var formPanel = this;
121         var fn = function(c){
122             if(c.doLayout && c != formPanel){
123                 Ext.applyIf(c, {
124                     labelAlign: c.ownerCt.labelAlign,
125                     labelWidth: c.ownerCt.labelWidth,
126                     itemCls: c.ownerCt.itemCls
127                 });
128                 if(c.items){
129                     c.items.each(fn);
130                 }
131             }else if(c.isFormField){
132                 f.add(c);
133             }
134         }
135         this.items.each(fn);
136     }
137 },'2.0');

162 /**
163  * Override the Ext.ComponentMgr so that whe can validate if a type is available
164  */
165 Ext.ComponentMgr = function(extMgr){
166     var types = {}; //List of all types
167 
168     function addTypes(condition,typeLine){
169       if (!condition || !typeLine) return;
170       for (var t=typeLine.split(','),i=0;i<t.length;i++) {types[t[i]]=t[i];}
171     }
172 
173     addTypes(Ext.isVersion('2.0'),'box,button,colorpalette,component,container,cycle,' +
174       'dataview,datepicker,editor,editorgrid,grid,paging,panel,progress,propertygrid,' +
175       'splitbutton,tabpanel,treepanel,viewport,window,toolbar,tbbutton,tbfill,tbitem,' +
176       'tbseparator,tbspacer,tbsplit,tbtext,form,checkbox,combo,datefield,field,fieldset,' +
177       'hidden,htmleditor,label,numberfield,radio,textarea,textfield,timefield,trigger');
178 
179     addTypes(Ext.isVersion('2.1'),'slider,statusbar');
180 
181     return Ext.applyIf({
182         registerType : function(xtype, cls){
183             extMgr.registerType(xtype,cls);
184             types[xtype] = xtype;
185         },
186 
187         isTypeAvailable : function (xtype) {
188           return !!types[xtype];
189         },
190 
191         allTypes : function(){
192           var arr = [];
193           for (var key in types) {
194             arr.push(key);
195           }
196           return arr;
197         }
198 
199     },extMgr);
200 }(Ext.ComponentMgr);
