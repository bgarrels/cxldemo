 20 //Register name spaces used
 21 Ext.namespace('Ext.ux.guid.data');  
 22 
 23 /**
 24  * CookieRepository extends a repository and can be used to save desginer content
 25  * in a cookie.
 26  */
 27 Ext.ux.guid.data.CookieRepository = Ext.extend(Ext.ux.guid.data.Repository,{
 28 
 29   /**
 30    * Init the cookieProvider
 31    */
 32   init : function(){
 33     this.cookies = new Ext.state.CookieProvider();     
 34     Ext.ux.guid.data.CookieRepository.superclass.init.call(this);
 35   },
 36   
 37   /**
 38    * Reload all items from the cookie
 39    * @param {function} callback A callback function called with parameter true or false
 40    * to indicated a succesfull refresh after refresh function is finished
 41    */
 42   refresh : function (callback) {
 43     this.items = this.cookies.get('repository.files');
 44     Ext.ux.guid.data.CookieRepository.superclass.refresh.call(this,callback);
 45   },
 46 
 47   /**
 48    * Save the changes made to a repositoryId
 49    * @param {String} id The repositoryId to use
 50    * @param {String} action The action to perform 'remove', 'new' and 'save'
 51    * @param {String} callback The callback function with parameter true or false
 52    * to indicated a succesfull action.
 53    * @param {String} content The json as String to used (optional)
 54    */
 55   saveChanges : function(id,action,callback,content) {  
 56     if (content) this.cookies.set('repository/' + id,escape(content));
 57     if (action=='remove') this.cookies.clear('repository/'+id);
 58     Ext.ux.guid.data.CookieRepository.superclass.saveChanges.call(this,id,action,callback,content);
 59     this.cookies.set('repository.files',this.items);
 60   },
 61 
 62   /**
 63    * Open a repository item by loading it and pasing the content back to the callback function
 64    * @param {String} id The repositoryId to use
 65    * @param {function} callback The callback function to use (boolean state,String content)
 66    * @param {String} content The content to use as default (optional)
 67    */
 68   open : function(id,callback,content) {
 69     content = unescape(this.cookies.get('repository/' + id)) || content;
 70     Ext.ux.guid.data.CookieRepository.superclass.open.call(this,id,callback,content);
 71   }
 72     
 73 });
