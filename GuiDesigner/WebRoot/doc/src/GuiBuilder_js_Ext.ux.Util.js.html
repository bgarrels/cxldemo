 80       * Retrun a time string in hh:mm:ss
 81       * @param {Date} curdate The date to used (default Date())
 82       * @return Date string in hh:mm:ss format

 84      dateStr : function(curdate) {
 85        curdate = curdate || new Date();
 86        return (curdate.getHours()<10 ? "0" : "") + curdate.getHours() + ':' +
 87               (curdate.getMinutes()<10 ? "0" : "") + curdate.getMinutes() + ':' +
 88               (curdate.getSeconds()<10 ? "0" : "") + curdate.getSeconds();

 91      /**
 92       * Convert a url into a nocache url by adding a date if needed
 93       * @param {String} url The url to use
 94       * @param {boolean} nocache The nocache variable (default this.nocache)
 95       */
 96      nocacheUrl : function(url,nocache){
 97        nocache = nocache==undefined ? this.nocache : nocache;
 98        if (nocache) {
 99          url += (url.indexOf('?') != -1 ? '&' : '?') + '_dc=' + (new Date().getTime());
100        }
101        return url;

104     /**
105      * A Synchronized Content loader for data from url
106      * @param {String} url The url to load synchronized
107      * @param {Boolean} nocache Should caching of file be disabled
108      * @param {Boolean} response Should response be returned instead of responseText
109      * @param {Object} param The params to be send
110      * @return {Boolean/String/XMLObject} When there was a error False otherwise response base on responseXML flag
111      */
112     syncContent : function(url,nocache,response,params) {
113      var activeX = Ext.lib.Ajax.activeX;
114      var isLocal = (document.location.protocol == 'file:');
115      var conn;
116 
117      try {
118       if(Ext.isIE7 && isLocal){throw("IE7forceActiveX");}
119       conn = new XMLHttpRequest();
120      } catch(e)  {
121        for (var i = 0; i < activeX.length; ++i) {
122          try {conn = new ActiveXObject(activeX[i]); break;} catch(e) {}
123        }
124      }
125      //Should we disable caching
126      url = this.nocacheUrl(url,nocache);
127      if(typeof params == "object") url  += (url.indexOf('?') != -1 ? '&' : '?') + Ext.urlEncode(params);
128      try {
129       conn.open('GET', url , false);
130       conn.send(null);
131       if ((isLocal && conn.responseText.length!=0) || (conn.status !== undefined && conn.status>=200 && conn.status< 300)) {
132         return response ? conn.response : conn.responseText;
133       }
134      } catch (e) {
135        this.fireEvent('error','synchronized',e);
136      }
137      return false;
138     },

140     /**
141      * Function used to load a JavaScript into a document.head element
142      * the id of the script item is the name of the file.
143      * @param {String} url The url to load javascript from
144      * @param {Boolean} nocache Should caching of javascript files be disabled
145      * @return {Boolean} Indicator if load went corretly true/false
146      */
147     scriptLoader : function(url,nocache) {
148      if (!url) return false;
149      var id=url;
150      if(!document.getElementById(id)) {
151        var content = this.syncContent(url,nocache);
152        if (content===false) return false;
153        var head = document.getElementsByTagName("head")[0];
154        var script = document.createElement("script");
155        try {
156          script.text = content;
157        } catch (e) {
158          script.appendChild(content);
159        }
160        script.setAttribute("type", "text/javascript");
161        script.setAttribute("id", id);
162        head.appendChild(script);
163      }
164      return true;
165     },

167     /**
168      * Get the value of a url action
169      * @param {String} name The name of the action
170      * @param {String} defaultValue The default value used when not found
171      * @param {String} url The url to use defaults to window url
172      * @return {String} The value found for action or default
173      */
174     getUrlAction : function (name,defaultValue,url) {
175       name = name.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
176       var regexS = "[\\?&]"+name+"=([^&#]*)";
177       var regex = new RegExp( regexS );
178       var results = regex.exec( url || window.location.href );
179       if( results == null )
180         return isNaN(defaultValue) ? defaultValue : Number(defaultValue);
181       else
182         return isNaN(results[1]) ? results[1] : Number(results[1]);
183     },

185     /**
186      * Parse a url into object
187      * containing element: "source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","name","ext","query","anchor"
188      * @param {String} str The url to parse
189      * @param {Object} options A object with options to be used during parse
190      */
191     parseUrl : function(str,options) {
192       var o   = Ext.applyIf(options || {},{
193           strictMode: false,
194           key: ["source","protocol","authority","userInfo","user","password","host","port","relative","path","directory","file","query","anchor"],
195           q:   {
196             name:   "queryKey",
197             parser: /(?:^|&)([^&=]*)=?([^&]*)/g
198           },
199           parser: {
200             strict: /^(?:([^:\/?#]+):)?(?:\/\/((?:(([^:@]*):?([^:@]*))?@)?([^:\/?#]*)(?::(\d*))?))?((((?:[^?#\/]*\/)*)([^?#]*))(?:\?([^#]*))?(?:#(.*))?)/,
201             loose:  /^(?:(?![^:@]+:[^:@\/]*@)([^:\/?#.]+):)?(?:\/\/)?((?:(([^:@]*):?([^:@]*))?@)?([^:\/?#]*)(?::(\d*))?)(((\/(?:[^?#](?![^?#\/]*\.[^?#\/.]+(?:[?#]|$)))*\/?)?([^?#\/]*))(?:\?([^#]*))?(?:#(.*))?)/
202           }
203         }),
204         m   = o.parser[o.strictMode ? "strict" : "loose"].exec(str),
205         uri = {},
206         i   = 14;
207 
208       while (i--) uri[o.key[i]] = m[i] || "";
209       uri[o.q.name] = {};
210       uri[o.key[12]].replace(o.q.parser, function ($0, $1, $2) {
211         if ($1) uri[o.q.name][$1] = $2;
212       });
213       if (!uri.protocol) {
214         uri.relative= uri.host + uri.relative;
215         uri.path = uri.host + uri.path;
216         uri.directory = uri.host + uri.directory;
217         uri.host="";
218       }
219       if (!uri.file) {
220         uri.file = uri.path;
221         uri.path = uri.directory = '';
222       }
223       uri['name']="";uri['ext']="";
224         if (/\.\w+$/.test(uri.file) && uri.file.match(/([^\/\\]+)\.(\w+)$/)) {
225           uri['name']=RegExp.$1;
226           uri['ext']=RegExp.$2;
227         }
228       return uri;
229     },

231     /**
232      * Load a url, stylesheet or javascript by adding it to the header
233      * @param {String} url The url to be added
234      * @param {String} type The load action to perform (default js)
235      * @param {Object} options options[reload] Should the file be reload if allready loaded
236      */
237     loadUrl : function(url,type,options) {
238       options = options || {};
239       //Now load it by adding it to the header
240       if (this._loadCount == undefined) {
241         this._loadCount=0;
242         this._loaded=[];
243       }
244       if (this._loaded[url] && !options.reload) return;
245       var node;
246       var self = this;
247       var head = document.getElementsByTagName("head")[0];
248       if(document.getElementById(url)){head.removeChild(url)}
249       switch (type) {
250        case 'css' :
251          node = document.createElement("link");
252          node.setAttribute("rel", "stylesheet");
253          node.setAttribute("type", "text/css");
254          node.setAttribute("href", this.nocacheUrl(url,options.nocache));
255          break;
256        default :
257          node = document.createElement("script");
258          node.setAttribute("type", "text/javascript");
259          node.setAttribute("src", this.nocacheUrl(url,options.nocache));
260       }
261       node.setAttribute("id",url);
262       node.onload = node.onreadystatechange = function() {
263        if (!self._loaded[url] && (!node.readyState || [4,"loaded","complete"].indexOf(node.readyState)!=-1)) {
264          self._loadCount--;
265          self._loaded[url]=true;
266        }
267       };
268       this._loadCount++;
269       this._loaded[url]=false;
270       head.appendChild(node);
271     },

273     /**
274      * Merge two json files into one, incase same items exists
275      * then second json is leading
276      * @param {Object\Array} json1 The first json file
277      * @param {Object\Array} json2 The second json file
278      * @return {Object\Array} The merged json
279      */
280     merge : function (item1,item2) {
281       if (item1 instanceof Array && item2 instanceof Array) {
282         var arr = [];
283         for (var i=0;i<item1.length;i++) arr.push(item1[i]);
284         for (var i=0;i<item2.length;i++) arr.push(item2[i]);
285         return arr;
286       } else if (typeof(item1)==typeof(item2) && typeof(item1)=='object') {
287         var obj = {};
288         for (var i in item1) {obj[i] = this.merge(item1[i],item2[i]);}
289         for (var i in item2) {if (!obj[i]) obj[i]=item2[i];}
290         return obj;
291       } else if (typeof(item1)==typeof(item2)) {
292         return item2;
293       } else if (item1==undefined) {
294         return item2;
295       } else if (item2==undefined) {
296         return item1;
297       } else {
298         throw new SyntaxError('Object items cannot be joined because items mismatch');
299       }
300     },

302     /**
303      * Transfer the form content into a object
304      * @param {Form} form The form to be transformed into a object
305      * @returns {Object} With the data of form
306      */
307     formObject : function(form){
308       if(typeof form == 'string') {
309           form = (document.getElementById(form) || document.forms[form]);
310       } else if (form instanceof Ext.form.FormPanel) {
311          form = form.getForm().getEl().dom;
312       } else if (form instanceof Ext.form.BasicForm) {
313          form = form.getEl().dom;
314       }
315 
316       var el, name, val, disabled, data, hasSubmit = false;
317       var setData = function(data,name,value) {
318         if (value!=undefined) { 
319            if (data==undefined) data = {};
320            if (data[name]!=undefined && data[name] instanceof Array) {
321              data[name].push(value);
322            } else if (data[name]!=undefined) {
323              data[name]=new Array(data[name]);
324              data[name].push(value);
325            } else data[name]=value;
326         }
327         return data;
328       };
329       for (var i = 0; form && i < form.elements.length; i++) {
330           el = form.elements[i];
331           disabled = form.elements[i].disabled;
332           name = form.elements[i].name;
333           val = form.elements[i].value;
334           if (!disabled && name){
335             switch (el.type)  {
336                   case 'select-one':
337                   case 'select-multiple':
338                       for (var j = 0; j < el.options.length; j++) {
339                           if (el.options[j].selected) {
340                              data =setData(data,name,Ext.isIE ?
341                                   (el.options[j].attributes['value'].specified ? el.options[j].value : el.options[j].text) :
342                                   (el.options[j].hasAttribute('value') ? el.options[j].value : el.options[j].text)
343                              );
344                           }
345                       }
346                       break;
347                   case 'radio':
348                   case 'checkbox':
349                       if (el.checked) {
350                           data = setData(data,name,val);
351                       }
352                       break;
353                   case 'file':
354                   case undefined:
355                   case 'reset':
356                   case 'button':
357                       break;
358                   case 'submit':
359                       if(hasSubmit == false) {
360                           data = setData(data,name,val);
361                           hasSubmit = true;
362                       }
363                       break;
364                   default:
365                       data =setData(data,name,val);
366                       break;
367               }
368           }
369       }
370       return data;
371     },

373     /**
374      * Function require is used to checks if the required files are loaded
375      * when not the files are loaded
376      * @param {Mixed} packages An array or ; seperated string with packages to load
377      * @param {Mixed} options An object with options to used, when string then basedir is assumed
378      * options[basedir] The default directory to use for all files
379      * options[cssdir] The directory to be used for stylesheets
380      * options[async] When set to true required will return directly
381      * options[callback] Callback function after all required files are loaded
382      * options[reload] When reload is set packages are reloaded
383      * options[nocache] When set the object caching is turned off
384      * options[reload] When true stylesheet/javacode is swapped
385      * @return {Object} a object with keys js and css contain an array with full path of items
386      */
387     require : function(packages,options){
388       var _require = function(packages,options){
389          var ret = {css: [],js:[]};
390          if (!packages) return ret;     
391          packages= (typeof(packages)=='string') ? packages.replace(',',';').split(';') : packages || [];
392          options = (typeof(options)=='string') ? {'basedir' : options} : options || {basedir : ""};
393          options['cssdir'] = options.cssdir || options.basedir;
394          options['reload'] = options['reload']== undefined ? true : options['reload'];
395          options['nocache'] = options['nocache']== undefined ? this.nocache : options['nocache'];
396          //First load all files
397          for (var i=0;i<packages.length;i++) {
398            //Create a name of path + packages + extentsion
399            var url = this.parseUrl(packages[i]);
400            if (['js','css'].indexOf(url.ext)==-1) url.file+='.js';
401            var dir = url.ext=='css' ? options.cssdir : options.basedir;
402            var uri = dir + (dir.length!=0 && dir.charAt(dir.length-1)!='/' ? "/" : "") +
403               url.directory +  url.file + (url.query ? '?' + url.query : '');
404            if (url.ext=='css') {
405                ret.css.push(uri);
406                Ext.util.CSS.swapStyleSheet(this.nocacheUrl(uri,options.nocache), uri);
407            } else if (options.async){
408               this.loadUrl(uri,url.ext,options);
409               ret.js.push(uri);
410            } else {
411               ret.js.push(uri);
412               this.scriptLoader(uri,options.nocache);
413            }
414          }
415         return ret;
416       }.createDelegate(this);
417       //Check if packages is an array if so walk it
418       if (packages instanceof Array && !options) {
419         var ret = {css: [],js:[]};
420         for (var i=0;i<packages.length;i=i+2) {
421           ret = this.merge(ret,_require(packages[i],packages[i+1]));
422         }
423         return ret;
424       } 
425       return _require(packages,options);
426     }
