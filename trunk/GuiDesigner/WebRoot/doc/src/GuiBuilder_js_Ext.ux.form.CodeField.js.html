 21 //Register name spaces used
 22 Ext.namespace('Ext.ux.form');
 23 
 24 /**
 25  * Component implementing a codefield. The codefield is a single line textarea using a 
 26  * triggerfield to open a full textarea or when install codepress.
 27  * @type component
 28  */
 29 Ext.ux.form.CodeField = Ext.extend(Ext.form.TriggerField,  {
 30     /**
 31      * The error to display when the code in the field is invalid (defaults to
 32      * '{value} is not a valid code - it must be in the format {format}').
 33      * @type {String} invalidText
 34      @cfg */
 35     invalidText : "'{0}' is not a valid code",
 36     /**
 37      * An additional CSS class used to style the trigger button.
 38      * @type {String} triggerClass
 39      @cfg */
 40     triggerClass : 'x-form-codefield-trigger',
 41 
 42     /**
 43      * Default language of scripteditor (defaults javascript)
 44      * @type {String}
 45      @cfg */
 46     language : 'javascript',
 47 
 48     /**
 49      * Should it use codePress as code editor (defaults true)
 50      * @type {Boolean}
 51      @cfg */
 52     codePress : true, //codePress enabled
 53 
 54     /**
 55      * Location of CodePress
 56      * @type {String}
 57      @cfg */
 58     codePressPath : undefined,
 59     
 60     /**
 61      * @private Change defualt Style
 62      */
 63     initComponent : function(){
 64       this.defaultAutoCreate = {
 65         tag: "textarea",
 66         rows : 1,
 67         style:"width:100px;height:1.8em;overflow:hidden;",
 68         autocomplete: "off", 
 69         wrap : "off"
 70       };
 71       Ext.ux.form.CodeField.superclass.initComponent.call(this);    
 72     },
 73 
 74     /**
 75      * InitEvents and redirect dblclick to triggerclick
 76      */
 77     initEvents : function(){
 78         Ext.ux.form.CodeField.superclass.initEvents.call(this);
 79         this.el.on('dblclick', this.onTriggerClick, this);
 80     },
 81 
 82     /**
 83      * Returns the current value of the  field
 84      * @return {String} value The value
 85      */
 86     getValue : function(){
 87       return Ext.ux.form.CodeField.superclass.getValue.call(this) || this.defaultValue || " ";
 88     },
 89 
 90     /**
 91      * Sets the value of the field.
 92      * @param {String} code The code string
 93      */
 94     setValue : function(code){
 95       Ext.ux.form.CodeField.superclass.setValue.call(this, this.formatCode(code));
 96       this.setCode(code);
 97     },
 98 
 99     /**
100      * Callback function when code is changed with setValue
101      * @param {String} code The code used
102      */
103     setCode : function(code) {
104     },
105 
106     /**
107      * Format the code, by default code is rtrimed
108      * @param {String} code The code string
109      * @returns {String} The formated code
110      */
111     formatCode : function(code) {
112      return String(code).replace(/\s+$/,""); //rtrim code
113     },
114 
115     // private
116     parseCode : function(value){
117       return true;
118     },
119 
120     //@private force reload of value after textarea is renderd when we have default value
121     initValue : function(){
122       Ext.ux.form.CodeField.superclass.initValue.call(this);
123       this.on('focus',function(){this.setValue(this.getValue())},this);
124     },
125 
126     //@private
127     validateValue : function(value){
128         if(!Ext.ux.form.CodeField.superclass.validateValue.call(this, value)){
129             return false;
130         }
131         if(value.length < 1){ // if it's blank and textfield didn't flag it then it's valid
132            this.setCode('');
133            return true;
134         }
135 
136         var parseOK = this.parseCode(value);
137 
138         if(!value || (parseOK == false)){
139             this.markInvalid(String.format(this.invalidText,value));
140             return false;
141         }
142         this.setCode(value);
143         return true;
144     },
145 
146     //@private Provides logic to override the default TriggerField.validateBlur which just returns true
147     validateBlur : function(){
148         return !this.editorWin || !this.editorWin.isVisible();
149     },
150 
151     //@private Implements the default empty TriggerField.onTriggerClick function to display the CodeField editor
152     onTriggerClick : function(){
153         if(this.disabled){
154             return;
155         }
156         if (!this.editorWin) {
157           var tf = (this.codePress && Ext.ux.CodePress)
158                  ?  new Ext.ux.CodePress({path: this.codePressPath, language: this.language ,autoResize:true,trim : true})
159                  :  new Ext.form.TextArea({autoCreate : {tag: "textarea",style:"width:160px;height:80px;",autocomplete: "off",wrap:"off"},resize:Ext.emptyFn});
160           this.editorWin = new Ext.Window({
161               title  : "CodeField Editor",
162               iconCls: 'icon-editEl',
163               closable:true,
164               width:600,
165               height:450,
166               plain:true,
167               modal: true,
168               maximizable : true,
169               layout      : 'fit',
170               items       : tf,
171               closeAction : 'hide',
172               keys: [{
173                   key: 27,
174                   scope: this,
175                   fn: function() {
176                     this.editorWin.hide();
177                     if (this.editor.cancelEdit) this.editor.cancelEdit();
178                   }}],
179               buttons: [{
180                  text    : "Close",
181                  scope   : this,
182                  handler : function() {
183                    this.editorWin.hide();
184                    if (this.editor.cancelEdit) this.editor.cancelEdit();
185                  }
186                 },{
187                  text    : "Apply",
188                  scope   : this,
189                  handler : function() {
190                    this.setValue(tf.getValue());
191                    this.editorWin.hide();
192                    this.editorWin.el.unmask();
193                    if (this.editor.completeEdit) this.editor.completeEdit();
194                  }
195                }]
196             });
197           this.editorWin.tf = tf;
198           this.editorWin.doLayout();
199           this.editorWin.on('resize',function () {tf.resize()});
200           }
201         this.editorWin.show();
202         this.editorWin.tf.setValue(this.getValue());
203     },
204 
205    //@private on render set the editor
206    onRender : function(ct, position){
207       this.editor = Ext.getCmp(ct.id) || {};
208       Ext.ux.form.CodeField.superclass.onRender.call(this, ct, position);
209    }
210 });
211 
212 Ext.reg('codefield',Ext.ux.form.CodeField);
