 21 Ext.namespace('Ext.ux.guid.tree');
 22 
 23 /**
 24  * Create a tree based on the content of a json file
 25  * When attr of json item is false a dynamic node is created
 26  * otherwhise a static tree node is created.
 27  * If json contains a list called childeren this list will be walked
 28  * to create childeren in tree
 29  * By default the treeNodes are not draggable
 30  */
 31 Ext.ux.guid.tree.JsonTreeLoader = Ext.extend(Ext.tree.TreeLoader,{
 32 
 33   //@private Should caching of pages be disabled
 34   nocache : false,
 35 
 36  /**
 37   * Create node but enabling childeren from Json
 38   * @param {object} attr The attributes from which to created nodes.
 39   * When attr.childeren also for these childeren nodes are created
 40   */
 41   createNode : function(attr){
 42     var childeren = attr.childeren;
 43     delete attr.childeren;
 44     if(this.baseAttrs){
 45         Ext.applyIf(attr, this.baseAttrs);
 46     }
 47     if(this.applyLoader !== false){
 48         attr.loader = this;
 49     }
 50     if(typeof attr.uiProvider == 'string'){
 51        attr.uiProvider = this.uiProviders[attr.uiProvider] || eval(attr.uiProvider);
 52     }
 53     if (!childeren) {
 54       return(attr.leaf===false ?
 55             new Ext.tree.AsyncTreeNode(attr) :
 56             new Ext.tree.TreeNode(attr) );
 57     } else {
 58       var node = new Ext.tree.TreeNode(Ext.applyIf(attr,{draggable:false}));
 59       var self = this;
 60       for(var i = 0, len = childeren.length; i < len; i++){
 61        if (Ext.isVersion(childeren[i].isVersion)) {
 62          if (childeren[i].wizard) { //Check if whe should create a wizard config
 63            childeren[i]['config'] = function(callback) {
 64              var w = new Ext.ux.JsonWindow({
 65                 x     : -1000, // Window is hidden by moving X out of screen
 66                 y     : -1000, //Window is hidden by moving Y out of screen
 67                 autoLoad : this.wizard,
 68                 callback : callback,
 69                 modal    : true,
 70                 nocache  : self.nocache 
 71              });
 72              w.json.on('error',function(type,exception){
 73                   Ext.Msg.alert('Wizard Load Error',type +" " + (typeof(exception)=='object' ? exception.message || exception : exception));
 74                   this.close();
 75                   return false;
 76                },w);
 77              w.show();
 78            }.createDelegate(childeren[i]);
 79          }
 80          var n = this.createNode(childeren[i]);
 81          if(n) node.appendChild(n);
 82        }
 83       }
 84       return node;
 85     }
 86   },
 87 
 88   /**
 89    * Allow dataUrl to be an  array of URLs to be loaded
 90    * @param {treeNode} node The treeNode for which to request the data
 91    * @param {function} callback The callback function
 92    */
 93   requestData : function(node, callback){
 94     if (this.dataUrl instanceof Array) {
 95        var allUrl = this.dataUrl;
 96        for (var i=0;i<allUrl.length;i++) {
 97          this.dataUrl = allUrl[i];
 98          Ext.ux.guid.tree.JsonTreeLoader.superclass.requestData.call(this,node,callback);
 99        }
100        this.dataUrl = allUrl;
101     } else {
102        Ext.ux.guid.tree.JsonTreeLoader.superclass.requestData.call(this,node,callback);
103     }
104   }
105 });
