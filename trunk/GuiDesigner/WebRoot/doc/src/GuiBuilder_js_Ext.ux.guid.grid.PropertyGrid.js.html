 23 /**
 24  * The propertyrecord used by the property grid.
 25  * Name is the key of a property, Value is the value of the property and
 26  * type is type of element. This type flag will control which editor is loaded
 27  */
 28  Ext.ux.guid.grid.PropertyRecord = Ext.data.Record.create([
 29     {name:'name',type:'string'}, 'value' , 'type'
 30 ]);
 31 
 32 /**
 33  * The constructor of property store
 34  */
 35 Ext.ux.guid.grid.PropertyStore = function(grid, source){
 36     Ext.ux.guid.grid.PropertyStore.superclass.constructor.call(this,grid,source);
 37     this.store = new Ext.data.Store({
 38         recordType : Ext.ux.guid.grid.PropertyRecord
 39     });
 40     this.store.on('update', this.onUpdate,  this);
 41 };
 42 
 43 /**
 44  * Constructor used to create a store used to store designer data
 45  */
 46 Ext.extend(Ext.ux.guid.grid.PropertyStore, Ext.grid.PropertyStore, {
 47     //@private The json id  
 48     jsonId : "__JSON__",
 49 
 50     /**
 51      * Get a property type for given name
 52      * @param {string} name The property to look fore
 53      */
 54     getPropertyType : function (name) {
 55       if (this.grid && this.grid.getPropertyType)
 56          return this.grid.getPropertyType(name);
 57       return null;
 58     },
 59 
 60     /**
 61      * protected  - should only be called by the grid.  Use grid.setSource instead.
 62      * Based on source object a property record is created
 63      * @param {object} o The source object 
 64      */
 65     setSource : function(o){
 66         this.source = o;
 67         this.store.removeAll();
 68         var data = [];
 69         for(var k in o){
 70             var orgKey = (k.indexOf(this.jsonId)==0 && k!=this.jsonId) ?
 71                  k.substring(this.jsonId.length) : null;
 72             if (orgKey && o[orgKey]==undefined) k =orgKey;
 73             if(k.indexOf(this.jsonId)!=0 && ['items'].indexOf(k)==-1){
 74                 var v = o[this.jsonId + k];
 75                 var type = null;
 76                 if (typeof(v)=='object') {
 77                    type = v.type || type;
 78                    v = v.display || v.value;
 79                 }
 80                 if (typeof(o[k]) == 'function') {
 81                   data.push(new Ext.grid.PropertyRecord({name: k, value: v || String(o[k]) , type : 'function'}, k));
 82                 } else if (typeof(o[k]) == 'object') {
 83                   data.push(new Ext.grid.PropertyRecord({name: k, value: v || String(Ext.ux.JSON.encode(o[k])), type : 'object'}, k));
 84                 } else {
 85                   data.push(new Ext.grid.PropertyRecord({name: k, value: v || o[k], type : type }, k));
 86                 }
 87             }
 88         }
 89         this.store.loadRecords({records: data}, {}, true);
 90     },
 91 
 92     /**
 93      * @private Update the data and fire beforepropetychange and propertychange events
 94      * @param {object} ds The datastore
 95      * @param {object} record The record
 96      * @param {String} The edit mode
 97      */
 98     onUpdate : function(ds, record, type){
 99         if(type == Ext.data.Record.EDIT){
100             var v = record.data['value'];
101             var oldValue = record.modified['value'];
102             if(this.grid.fireEvent('beforepropertychange', this.source, record.id, v, oldValue) !== false){
103                 record.data.changeValue = this.updateSource(record.id,v,record.data.type);
104                 record.commit();
105                 this.grid.fireEvent('propertychange', this.source, record.id, v, oldValue);
106             }else{
107                 record.reject();
108             }
109         }
110     },
111 
112     /**
113      * @private Update a source value property
114      * @param {String} key The property key to change
115      * @param {String} value The value to change
116      * @param {String} type The property type
117      */
118     updateSource : function (key,value,type) {
119       var propType = this.getPropertyType(key);
120       if (!type && propType) type=propType.type;
121       if (this.grid.fireEvent('propertyvalue',this.source,key,value,type,propType)) {
122         this.source[key] = value;
123       }
124       return this.source[key];
125     },
126 
127     /**
128      * Set the value of a property
129      * @param {String} prop The property to set
130      * @param {String} value The value to be set
131      */
132     setValue : function(prop, value){
133         this.store.getById(prop).set('value', value);
134         this.updateSource(prop,value);
135     }
136 
137 });

140 /**
141  * Constructior which extends default PropertyColumnModel in so that 
142  * we implement other editors
143  */
144 Ext.ux.guid.grid.PropertyColumnModel = function(grid, store){
145     Ext.ux.guid.grid.PropertyColumnModel.superclass.constructor.call(this,grid,store);
146     this.setConfig( [
147         {header: this.nameText, width:40, resizable:true, sortable: true, dataIndex:'name', id: 'name', menuDisabled:true},
148         {header: this.valueText, width:60, resizable:false, dataIndex: 'value', id: 'value', menuDisabled:true}
149     ]);
150     this.jsonId=grid.jsonId;
151     Ext.apply(this.editors,{
152         'regexp' : new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'new RegExp()',codePress:grid.codePress,codePressPath:grid.codePressPath})),
153         'function':new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'function(){}',codePress:grid.codePress,codePressPath:grid.codePressPath})),
154         'object':new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'{}',codePress:grid.codePress,codePressPath:grid.codePressPath})),
155         'object/array':new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'[{}]',codePress:grid.codePress,codePressPath:grid.codePressPath})),
156         'array': new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'[]',codePress:grid.codePress,codePressPath:grid.codePressPath})),
157         'template': new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'',codePress:grid.codePress,codePressPath:grid.codePressPath})),
158         'mixed': new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'',codePress:grid.codePress,codePressPath:grid.codePressPath})),
159         'html' : new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'',language:'html',codePress:grid.codePress,codePressPath:grid.codePressPath})),
160         'css' : new Ext.grid.GridEditor(new Ext.ux.form.CodeField({defaultValue:'',language:'css',codePress:grid.codePress,codePressPath:grid.codePressPath})),
161         'editlist' :new Ext.grid.GridEditor(new Ext.ux.form.SimpleCombo({forceSelection:false,data:[],editable:true,customProperties:true})),
162         'list':new Ext.grid.GridEditor(new Ext.ux.form.SimpleCombo({forceSelection:false,data:[],editable:true,customProperties:false})),
163         'boolean':new Ext.grid.GridEditor(new Ext.ux.form.SimpleCombo({forceSelection:false,data:[[true,'true'],[false,'false']],editable:true,customProperties:true}))
164     });
165     this.valueRendererDelegate = this.valueRenderer.createDelegate(this);
166     this.propertyRendererDelegate = this.propertyRenderer.createDelegate(this);
167 };

169 /**
170  * Extend the default PropertyColumnModel in so that we implement other editors
171  */
172 Ext.extend(Ext.ux.guid.grid.PropertyColumnModel,Ext.grid.PropertyColumnModel, {
173     /**
174      * Get a property type for given name
175      * @param {string} name The property to look fore
176      */  
177     getPropertyType : function (name) {
178       if (this.grid && this.grid.getPropertyType) return this.grid.getPropertyType(name);
179       return null;
180     },
181 
182     /**
183      * Get the celleditor by row and colIndex
184      * @param {int} colIndex The column index
185      * @param {int} rowIndex The row index
186      * @returns {Object} The cellEditor
187      */
188     getCellEditor : function(colIndex, rowIndex){
189         var p = this.store.getProperty(rowIndex);
190         var n = p.data['name'], val = p.data['value'], t = p.data['type'];
191         if(this.grid.customEditors[n]){
192             return this.grid.customEditors[n];
193         }
194         var prop = this.getPropertyType(n);
195         //Check if there is a property and if there is a editor
196         if (prop) {
197           if (prop.editor) {
198             if (typeof(prop.editor)!='string') {
199                return prop.editor;
200             }
201             t = prop.editor;
202           }
203           t = t || prop.type;
204           if (!t && prop.values) {
205             var editor =  prop.editable ? this.editors['editlist'] : this.editors['list'];
206             editor.field.setList(prop.values);
207             return editor;
208           }
209         }
210 
211         if (t && this.editors[t]) {
212           return this.editors[t];
213         } else if(Ext.isDate(val)){
214             return this.editors['date'];
215         }else if(typeof val == 'number'){
216             return this.editors['number'];
217         }else if(typeof val == 'boolean'){
218             return this.editors['boolean'];
219         }
220         return this.defaultEditor || this.editors[prop ? 'string' : 'mixed'];
221     },
222 
223     /**
224      * Format the value of cell
225      * @param {object} The value to be formated
226      * @param {Element} The element
227      * @param {Record} The record
228      */
229     valueRenderer  : function(value, p, r) {
230       if (typeof value == 'boolean') {
231         p.css = (value ? "typeBoolTrue" : "typeBoolFalse");
232         return (value ? "True" : "False");
233       }
234       var propType = this.getPropertyType(r.id);
235       if (propType && ['object','array','object/array'].indexOf(propType.type)!=-1) {
236         p.css = "typeObject";
237       }
238       if (typeof(value)=="string" && value.length>24) {
239         value = value.substring(0,21) + "...";
240       }
241       //Decode the value for html
242       var div = document.createElement('div');
243       div.appendChild(document.createTextNode(value));
244       return div.innerHTML;
245     },
246 
247     /**
248      * @private Implement a tooltip
249      */
250     propertyRenderer :  function(value, p) {
251       var propType = this.getPropertyType(value);
252       if (propType) {
253         qtip = propType.desc || '';
254         p.attr = 'qtip="' + qtip.replace(/"/g,'"') + '"'; //'
255       }
256       return value;
257     },
258 
259     /**
260      * @private Return one of the property renders
261      */
262     getRenderer  : function(col){
263        return col == 0 ? this.propertyRendererDelegate : this.valueRendererDelegate;
264     }
265 });

268 /**
269  * Create a property gird by extending a Grid
270  * @type component
271  */
272 Ext.ux.guid.grid.PropertyGrid = Ext.extend(Ext.grid.EditorGridPanel, {
273     // @private columns cannot be moved
274     enableColumnMove:false,
275     // @private Rows will not be stripped
276     stripeRows:false,
277     // @private Mouse over events are not tracked
278     trackMouseOver: false,
279     // @private Single click will start edit
280     clicksToEdit:1,
281     // @private noHD menu
282     enableHdMenu : false,
283     // @private Grid is forced to fit space
284     viewConfig : {forceFit:true},
285     // @private The internal jsonId used
286     jsonId : '__JSON__',
287 
288     /**
289      * Should it use codePress as code editor (defaults true)
290      * @type {Boolean}
291      @cfg */
292     codePress : true, //codePress enabled
293 
294     /**
295      * Location of CodePress
296      * @type {String}
297      @cfg */
298     codePressPath : undefined,
299 
300     /**
301      * Get a property type for given name
302      * @param {string} name The property to look fore
303      */   
304     getPropertyType : function (name) {
305       if (this.propertyTypes) {
306         var i = this.propertyTypes.find('name',name);
307         if (i!=-1) return this.propertyTypes.getAt(i).data;
308       }
309       return null;
310     },
311 
312     /**
313      * @private The component and events
314      */
315     initComponent : function(){
316         this.customEditors = this.customEditors || {};
317         this.lastEditRow = null;
318         var store = new Ext.ux.guid.grid.PropertyStore(this);
319         store.jsonId=this.jsonId,
320         this.propStore = store;
321         var cm = new Ext.ux.guid.grid.PropertyColumnModel(this, store);
322         store.store.sort('name', 'ASC');
323         this.addEvents(
324             /**
325               * Event fired before value changes
326               * @param {Source} source The source object
327               * @param {String} key The key changed
328               * @param {String} value The value to be set
329               * @param {String} type The type of value
330               * @return {Boolean} False will assume custom assignment
331               */
332             'beforepropertychange',
333             /**
334              * Event fired when propery changed
335              * @param {Source} source The source object
336              * @param {String} key The key changed
337              * @param {String} value The value to be set
338              * @param {String} type The type of value
339              * @return {Boolean} False will assume custom assignment
340              */
341             'propertychange',
342             /**
343              * Event fired to allow custom change of value
344              * @param {Source} source The source object
345              * @param {String} key The key changed
346              * @param {String} value The value to be set
347              * @param {String} type The type of value
348              * @return {Boolean} False will assume custom assignment
349              */
350             'propertyvalue'
351         );
352         this.cm = cm;
353         this.ds = store.store;
354         Ext.ux.guid.grid.PropertyGrid.superclass.initComponent.call(this);
355 
356         this.selModel.on('beforecellselect', function(sm, rowIndex, colIndex){
357             if(colIndex === 0){
358                 this.startEditing.defer(200, this, [rowIndex, 1]);
359                 return false;
360             }
361         }, this);
362     },
363 
364     /**
365      * @private Apply the property grid style
366      */
367     onRender : function(){
368         Ext.ux.guid.grid.PropertyGrid.superclass.onRender.apply(this, arguments);
369         this.getGridEl().addClass('x-props-grid');
370     },
371 
372     /**
373      * @private After render the source is set
374      */
375     afterRender: function(){
376         Ext.ux.guid.grid.PropertyGrid.superclass.afterRender.apply(this, arguments);
377         if(this.source){
378             this.setSource(this.source);
379         }
380     },
381 
382     /**
383      * Set the source of the grid
384      * @param {Object} The source object
385      */
386     setSource : function(source){
387         this.propStore.setSource(source);
388     },
389 
390     /**
391      * Get the datasource of the grid
392      * @returns {Object} The datasource object
393      */
394     getSource : function(){
395         return this.propStore.getSource();
396     }
397 });
